## vManage Performance Metrics Access Verification
## Loop through the temp-vmanage-main-cust-variable-dataset & verify access to each vManage
## source_ip and collection_timestamp come from query-api; add-missing-columns ensures stream write does not fail when API omits them
--> @c:data-loop dataset="temp-vmanage-main-cust-variable-dataset" and columns="name,customer_tag,customer_id"
    --> @$name:query-api path = 'device' and continue_on_failure = 'yes'
    --> @dm:add-missing-columns columns="collection_status,reason,customer_id,customer_tag,source_ip,collection_timestamp"
    --> @dm:eval customer_id = "'$customer_id'"
    --> @dm:eval customer_tag = "'$customer_tag'"
    --> *exec:if-condition collection_status != 'Success'
       --> @dm:map to = 'collection_timestamp' & func = 'time_now'
       --> @dm:dedup columns="source_ip"
       --> *dm:safe-filter * get source_ip as 'asset_id',collection_status,collection_timestamp,reason,customer_id,customer_tag
       --> @dm:eval bot_source_name = "'$name'"
       --> @dm:eval bot_source_type = "'cisco-vmanage'"
       --> @dm:eval bot_name = "'query-api device'"
       --> @dm:eval customer_id = "'' if customer_id == 'Not Available' else customer_id"
       --> @dm:eval customer_tag = "'' if customer_tag == 'Not Available' else customer_tag"
       
    --> @exec:end-if
   --> *exec:if-condition collection_status = 'Success'
      --> @dm:eval vmanage_name = "'$name'"
      --> @dm:save name = 'temp-vmanage_device_inventory' & append = 'yes'
      --> @dm:dedup columns = 'source_ip'
      --> *dm:safe-filter * get source_ip as 'asset_id',collection_status,collection_timestamp,reason,customer_id,customer_tag
      --> @dm:eval bot_source_name = "'$name'"
      --> @dm:eval bot_source_type = "'cisco-vmanage'"
      --> @dm:eval bot_name = "'query-api device'"
      --> @dm:eval customer_id = "'' if customer_id == 'Not Available' else customer_id"
      --> @dm:eval customer_tag = "'' if customer_tag == 'Not Available' else customer_tag"
      --> @dm:save name="vmanage_perf_access_verification_status-$loop_index"
   --> @exec:end-if

## Concat all the vManage responses to a dataset
--> @c:new-block
    --> @dm:concat names="vmanage_perf_access_verification_status-.*"
    --> @dm:save name="vmanage_perf_access_verification_status"

## Write to stream
--> @c:new-block
    --> @dm:recall name="vmanage_perf_access_verification_status"
    --> @dm:skip-block-if-shape row_count="0"
    --> @dm:copy-columns from="collection_timestamp" and to="timestamp"
    --> @dm:eval customer_id = "'' if customer_id == 'Not Available' else customer_id"
    --> @dm:eval customer_tag = "'' if customer_tag == 'Not Available' else customer_tag"
    --> @rn:write-stream name="vmanage_access_verification_status_stream"
