       --> @dm:save name="temp-input_ds_to_generate_bucket_id"
       --> @dm:selectcolumns include="parent_serial_number"
       --> @dm:copy-columns from = "parent_serial_number" and to = "parent_identifier" and func = "strip"
       --> @dm:add-checksum checksum_type="rows-only" and row_checksum_column="parent_identifier" and key="yes"
       ## Example: hash="abc123def456..." -> hexa_tail="f456"
       ## If hash is shorter than 4 chars, default to '0000'
       --> @dm:eval hexa_tail = "parent_identifier[-4:] if len(parent_identifier) >= 4 else '0000'"
       ## Example: hexa_tail="f456" -> hexa_int=62550
       --> @dm:eval hexa_int = "int(hexa_tail, 16)"
       ## Result is always between 0 and 15
       ## Example: hexa_int=62550 -> bucket_id=62550%16=6
       --> @dm:eval bucket_id = "hexa_int % int(16)"
       ## Save the bucket ID mapping (parent_serial_number -> bucket_id, parent_identifier)
       --> @dm:save name="temp-unique_id_with_bucket_id"
       --> @dm:recall name="temp-input_ds_to_generate_bucket_id"
       ## Enrich the original dataset with bucket_id and parent_identifier columns
       ## Match on parent_serial_number to add bucket_id and parent_identifier to each row
       --> @dm:enrich dict="temp-unique_id_with_bucket_id" and
               src_key_cols="parent_serial_number" and
               dict_key_cols="parent_serial_number" and
               enrich_cols="bucket_id,parent_identifier"
    --> @exec:end-function

## Function: ignore_assets_in_states
## Purpose: Define a list of asset states that should be excluded from processing
## This function creates a configuration table of asset states with their exclusion rules
## Returns only the states that have enabled='true' (i.e., states that should be ignored)
## 
## Configuration:
## - OFFLINE in MANUAL mode: Enabled (ignore these assets)
## - MAINTENANCE in MANUAL mode: Enabled (ignore these assets)
## - DECOMMISSION in MANUAL mode: Enabled (ignore these assets)
## - DECOMMISSION in AUTO mode: Disabled (do NOT ignore these assets)
--> @c:flex-block
    --> @dm:empty
    --> @exec:define-function name= "ignore_assets_in_states"
       --> @dm:empty
       ## Add rule: OFFLINE state in MANUAL mode should be ignored
       --> @dm:addrow asset_status='OFFLINE' and enabled='true' and asset_mode='MANUAL'
       ## Add rule: MAINTENANCE state in MANUAL mode should be ignored
       --> @dm:addrow asset_status='MAINTENANCE' and enabled='true' and asset_mode='MANUAL'
       ## Add rule: DECOMMISSION state in MANUAL mode should be ignored
       --> @dm:addrow asset_status='DECOMMISSION' and enabled='true' and asset_mode='MANUAL'
       ## Add rule: DECOMMISSION state in AUTO mode should NOT be ignored (enabled='false')
       ## This allows AUTO-decommissioned assets to be processed
       --> @dm:addrow asset_status='DECOMMISSION' and enabled='false' and asset_mode='AUTO'
    --> @exec:end-function

## Function: exclude_assets_by_state
## Purpose: Filter out assets that should be excluded based on their state and mode
## This function processes incoming assets, identifies which ones should be ignored based on
## state exclusion rules, and writes only the valid assets to the persistent stream
## 
## Flow:
## 1. Validate input is not empty
## 2. Separate stream assets (existing in database) from new assets to process
## 3. Get exclusion rules for asset states
## 4. Match existing stream assets against exclusion rules to identify ignored assets
## 5. Remove ignored assets from the new asset dataset
## 6. Write remaining assets to the persistent stream
--> @c:flex-block
    --> @dm:empty
    --> @exec:define-function name= "exclude_assets_by_state"
       --> @dm:save name="temp-func_input"
       --> *exec:if-shape num_rows = 0
          --> @exec:return-function
       --> @exec:end-if
       ## STEP 1: Separate Stream Assets from New Assets
       ## Get the parent assets data that was fetched from stream and passed to this block
       ## These are existing assets already in the persistent stream (marked with is_stream_asset='yes')
       ## This data is used to check which existing assets should be ignored
       --> @dm:save name="temp-db_assets_data"
       ## STEP 2: Get Asset State Exclusion Rules
       ## Invoke the ignore_assets_in_states function to get the list of states/modes to ignore
       --> @exec:call-function name = "ignore_assets_in_states"
       ## Filter to return only enabled exclusion rules
       ## This returns the list of state/mode combinations that should trigger asset exclusion
       --> *dm:filter enabled is 'true'
       --> @dm:save name="temp-ignore_asset_states"
       ## STEP 3: Identify Assets to Ignore
       ## Match existing stream assets against exclusion rules to find which ones should be ignored
       ## Load the exclusion rules (only enabled rules)
       ## Enrich stream assets with unique_id if they match exclusion rules
       ## Match on asset_status and asset_mode to identify ignored assets
       ## This will add unique_id column only for assets that match exclusion criteria
       --> @dm:enrich dict="temp-db_assets_data" and
               src_key_cols="asset_status,asset_mode" and
               dict_key_cols="asset_status,asset_mode" and
               enrich_cols="unique_id,parent_serial_number" and
               dedup_dict="no"
       ## Filter to get only assets that matched exclusion rules (have unique_id populated)
       --> *dm:filter unique_id is not null
       ## Mark these assets as matched (will be ignored)
       --> @dm:eval matched='"yes"'
       ## Select only unique_id and matched columns to create ignored assets list
       --> @dm:selectcolumns include='unique_id|parent_serial_number|matched'
       --> @dm:save name="temp-ignored_assets"
    --> @exec:end-function
